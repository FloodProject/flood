Input Handling

	Explain its purpose and how to use it.
	
	InputMap inputMap( *getInputManager() );
	
	// Action callback binding
	inputMap["Run"]->bind( &Example::Run, this );

	// Manual action trigger binding
	inputMap.register( "Run", Keys::W );

	// Automatic action trigger binding
	inputMap.load( "actions.json" );
